Conditionals

Switch/case with dictionary mapping

It's possible to write workarounds of "switch/case" statements on Python. One of these would be the use of a dictionary mapping:

def numbers_to_strings(argument):
    switcher = {
        0: "zero",
        1: "one",
        2: "two",
    }
    return switcher.get(argument, "nothing")

This code is analogous to:

function(argument){
    switch(argument) {
        case 0:
            return "zero";
        case 1:
            return "one";
        case 2:
            return "two";
        default:
            return "nothing";
    };
};

Switch/case with dictionary mapping for functions

Another approach would be the use of dictionary mapping for functions. In Python, we can also include functions or lambdas in our dictionary mapping:

def zero():
    return "zero"

def one():
    return "one"

def numbers_to_functions_to_strings(argument):
    switcher = {
        0: zero,
        1: one,
        2: lambda: "two",
    }
    # Get the function from switcher dictionary
    func = switcher.get(argument, lambda: "nothing")
    # Execute the function
    return func()

While the code inside zero() and one() are simple, many Python programs use dictionary mappings like this to dispatch complex procedures.

Switch/case with dispatch methods for classes

A different approach would be the use of dispatch methods for classes. In Python, if we don't know what method to call on a class, we can use a dispatch method to determine it at runtime.

class Switcher(object):
    def numbers_to_methods_to_strings(self, argument):
        """Dispatch method"""
        # prefix the method_name with 'number_' because method names
        # cannot begin with an integer.
        method_name = 'number_' + str(argument)
        # Get the method from 'self'. Default to a lambda.
        method = getattr(self, method_name, lambda: "nothing")
        # Call the method as we return it
        return method()

    def number_0(self):
        return "zero"

    def number_1(self):
        return "one"

    def number_2(self):
        return "two"

Strings

Let's see a brief example section for working with strings:

>>> s = "Hello, World! This is Python"

# Basic operators
>>> 'h' in s    # Check if 'h' is in 's'
True
>>> 'A' in s    # Check if 'A' is in 's'
False
>>> s + ", Peter"    # Concatenate strings
'Hello, World! This is Python, Peter'
>>> " Hello" * 3    # Replicate strings
' Hello Hello Hello'

# Built-in functions
>>> len(s)    # Length or amount of characters in 's'
28
>>> min(s)    # Minimum value in 's' (ASCII value of <Space> is 32d)
' '
>>> max(s)    # Maximum value in 's' (ASCII value of 'y' is 121d)
'y'

# "Find" method
>>> s.find('h')    # Try to find the position of the first ocurrence of 'h' in 's'
15    # s[15] = 'h'
>>> s.find('A')
-1    # (not found)
>>> s.find('o', 5)    # Find first 'o' after position [5] in s
8    # s[8] = 'o'
>>> s.find('o', 9, 27)    # Find first 'o' between [9] and [27]
26    # s[26] = 'o'

# "Split" method
>>> s.split()    # Split 's' whenever <Space> is found. Return a list
['Hello,', 'World!', 'This', 'is', 'Python']
>>> s.split('!')    # Split 's' whenever '!' is found. Return a list
['Hello, World', ' This is Python']

# "Join" method
>>> i = ['Hello,', 'World!', 'This', 'is', 'Python']
>>> " ".join(x)    # Joins the list items into a string whenever ' ' is found
'Hello, World! This is Python'
>>> j = ['Hello, World', ' This is Python']
>>> "!".join(j)
'Hello, World! This is Python'

# "Lower" and "upper" methods
>>> s.lower()
'hello, world! this is python'
>>> s.upper()
'HELLO, WORLD! THIS IS PYTHON'

# "Strip" method
>>> a = "   " + s + ", Peter *!@   "
>>> a.strip(' *!@')    # Removes leading and trailing whitespace or characters
'Hello, World! This is Python, Peter'

# "Replace" method
>>> s.replace('World', 'Mary')    # Replaces substring 'World' for 'Mary'
'Hello, Mary! This is Python'

# Useful constants
>>> import string
>>> string.digits
'0123456789'
>>> string.hexdigits
'0123456789abcdefABCDEF'
>>> string.octdigits
'01234567'
>>> string.ascii_letters
'abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ'
>>> string.ascii_lowercase
'abcdefghijklmnopqrstuvwxyz'
>>> string.ascii_uppercase
'ABCDEFGHIJKLMNOPQRSTUVWXYZ'
>>> string.printable
'0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ!"#$%&\'()*+,-./:;<=>?@[\\]^_`{|}~ \t\n\r\x0b\x0c'
>>> string.punctuation
'!"#$%&\'()*+,-./:;<=>?@[\\]^_`{|}~'
>>> string.whitespace
' \t\n\r\x0b\x0c'

Lists

It is important to notice that indexing can be done from the front using positive indices (starting with 0 as the first element) and from the back using negative indices (starting with -1 as the last element). Let's see an example of this:

>>> x = ["first", "second", "third", "fourth"]
>>> x[0]
'first'
>>> x[-1]
'fourth'
>>> x[2]
'third'
>>> x[-2]
'third'

Let's see a brief example section for working with lists:

>>> names = [ "Peter", "Dalia", "Johnny", "Rick" ]

# Basic operators
>>> 'Dalia' in names    # Check if 'Dalia' is in 'names'
True
>>> 'Terry' in names    # Check if 'Terry' is in 'names'
False
>>> names + ["Jasmin", "Philys", "Ella"]    # Concatenate lists
['Peter', 'Dalia', 'Johnny', 'Rick', 'Jasmin', 'Philys', 'Ella']
>>> ["Jasmin", "Philys"] * 4    # Replicate lists
['Jasmin', 'Philys', 'Jasmin', 'Philys', 'Jasmin', 'Philys', 'Jasmin', 'Philys']

# Built-in functions
>>> len(names)    # Length or amount of items in 'names'
4
>>> min(names)    # Minimum value in 'names' (ASCII value of first letter 'D')
'Dalia'
>>> max(names)    # Maximum value in 'names' (ASCII value of first letter 'R')
'Rick'
>>> del names[1]    # Delete element in position[1] ("Dalia")
>>> names
['Peter', 'Johnny', 'Rick']
>>> del names[-1]    # Delete element in position[-1] ("Rick")
>>> names
['Peter', 'Johnny']

# "Append" method
>>> names.append("Brian")    # Append name "Brian" at the end of 'names'
>>> names
['Peter', 'Johnny', 'Brian']

# "Count" method
>>> names.count("Johnny")    # Counts ocurrences of "Johnny" in 'names'
1
>>> names.count("Josh")    # Counts ocurrences of "Josh" in 'names'
0

# "Extend" method
>>> names.extend(["Trinity", "Carly"])    # Appends a new list at the end of 'names'
>>> names                             
['Peter', 'Johnny', 'Brian', 'Trinity', 'Carly']

# "Index" method
>>> names.index("Carly")    # Returns the smallest 'i' where s[i] == "Carly"
4
>>> names.index("Carly", 0, 5)    # Returns the position of 'Carly' between '0, 5'
4

# "Insert" method
>>> names.insert(3, "Daniel")
>>> names
['Peter', 'Johnny', 'Brian', 'Daniel', 'Trinity', 'Carly']

# "Pop" method (used to implement a data structure called stack)
>>> names.pop()    # Returns the last element
'Carly'
>>> names
['Peter', 'Johnny', 'Brian', 'Daniel', 'Trinity']
>>> names.pop(1)    # Returns the [1] element 
'Johnny'
>>> names
['Peter', 'Brian', 'Daniel', 'Trinity']
>>> names.pop(0)    # Returns the first element
'Peter'
>>> names
['Brian', 'Daniel', 'Trinity']

# "Remove" method
>>> names.remove("Brian")    # Searches for "Brian" and removes it from 'names'
>>> names
['Daniel', 'Trinity']

# "Reverse" method
>>> names = [ "Peter", "Dalia", "Johnny", "Rick" ]
>>> names.reverse()    # Reverses items of 'names' in place
>>> names
['Rick', 'Johnny', 'Dalia', 'Peter']

# "Sort" method
>>> names.sort()    # Arrange items of 'names' in place
>>> names
['Dalia', 'Johnny', 'Peter', 'Rick']
>>> names.sort(reverse=True)    # Arrange items of 'names' in place in reverse order
>>> names
['Rick', 'Peter', 'Johnny', 'Dalia']

Tuples

Tuples are similar to lists but are immutable. That is, they can't be modified after they have been created. There are also only two tuple methods: count and index. A major purpose of tuples is for use as keys for dictionaries. They're also more efficient to use when you don't need modifiability.

Let's see a brief example section for working with tuples:

>>> names = ("Peter", "Dalia", "Johnny", "Rick")

# Basic operators
>>> 'Dalia' in names    # Check if 'Dalia' is in 'names'
True
>>> 'Terry' in names    # Check if 'Terry' is in 'names'
False
>>> names + ("Jasmin", "Philys", "Ella")    # Concatenate tuples
('Peter', 'Dalia', 'Johnny', 'Rick', 'Jasmin', 'Philys', 'Ella')
>>> ("Jasmin", "Philys") * 4    # Replicate tuples
('Jasmin', 'Philys', 'Jasmin', 'Philys', 'Jasmin', 'Philys', 'Jasmin', 'Philys')

# Built-in functions
>>> len(names)    # Length or amount of items in 'names'
4
>>> min(names)    # Minimum value in 'names' (ASCII value of first letter 'D')
'Dalia'
>>> max(names)    # Maximum value in 'names' (ASCII value of first letter 'R')
'Rick'

# "Count" method
>>> names.count("Johnny")    # Counts ocurrences of "Johnny" in 'names'
1
>>> names.count("Josh")    # Counts ocurrences of "Josh" in 'names'
0

# "Index" method
>>> names.index("Peter")    # Returns the smallest 'i' where s[i] == "Peter"
0
>>> names.index("Peter", 0, 5)    # Returns the position of 'Peter' between '0, 5'
4
